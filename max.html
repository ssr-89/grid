Grid or Flex?

HTML
<div class="parent">
  <div class="child"></div>
  <div class="child"></div>
  <div class="child"></div>
  <div class="child"></div>
  <div class="child"></div>
  <div class="child"></div>
  <div class="child"></div>
  <div class="child"></div>
</div>

CSS
// parent - родительский блок
.parent {
  display: grid;
  grid-template-columns: repeat(12, 1fr); // 12 колонок
  gap: 23px;
}

// child - дочерний блок
.child {
  grid-column: span 3; // занимает 3 колонки из 12

  min-height: 300px;
  background-color: aquamarine;
}

// брейкпоинты
@media (max-width: 991.98px) {
  .parent {
    grid-template-columns: repeat(6, 1fr);
  }
}
@media (max-width: 479.98px) {
  .parent {
    grid-template-columns: repeat(3, 1fr);
  }
}
